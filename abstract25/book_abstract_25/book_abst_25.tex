\newenvironment{abstract_online}[4] % {title}{author}{affiliation}{type}
{
  \phantomsection
 % Unnumbered section with just the title visible in the document
  \addcontentsline{toc}{section}{\textbf{#1} \textit{-- #2}} % Add to TOC WITHOUT number
	\begin{center}
	\noindent{\normalsize \textbf{#1}}\par\vspace{0.3em}
    \noindent{\normalsize \itshape #2}\par\vspace{0.3em}  % Author in italics
    \noindent{\small \textsc{#3}}\par\vspace{0.5em}  % Affiliation in small caps
    \noindent{\footnotesize #4}
	\end{center}
  \vspace{0.5em}
}
